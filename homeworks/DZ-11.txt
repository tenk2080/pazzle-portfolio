using System;

namespace KB5
{
    public class Book
    {
        public string Name { get; set; }
        public string Autor { get; set; }
        public int Page { get; set; }

        public Book() { }

        public Book(string name)
        {
            Name = name;
        }

        public Book(string name, string autor)
        {
            Name = name;
            Autor = autor;
        }

        public Book(string name, string autor, int page)
        {
            Name = name;
            Autor = autor;
            Page = page;
        }

        public void PrintInfo()
        {
            Console.WriteLine($"Книга: {Name}, Автор: {Autor}, Страниц: {Page}");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            var book = new Book();
            var book1 = new Book("Lev");
            var book2 = new Book("Lev", "rq");
            var book3 = new Book("Lev", "rq", 1000);

            book.PrintInfo();
            book1.PrintInfo();
            book2.PrintInfo();
            book3.PrintInfo();
        }
    }
}
